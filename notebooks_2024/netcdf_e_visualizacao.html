# converter coordenadas para o clabel:
pontos_x=np.asarray([10,22,20])
pontos_y=np.asarray([-7,-2,-15])
pp=m.transform_points(ccrs.PlateCarree(),pontos_x,pontos_y)
pontos_x_proj=pp[:,0]
pontos_y_proj=pp[:,1]


pl.clabel(c,manual=zip(pontos_x_proj,pontos_y_proj),colors='k',inline_spacing=0.5,fontsize=8,fmt='%d')

import cartopy.crs as ccrs
import cartopy

fig=pl.figure()
fig.set_size_inches([9,7])
m=ccrs.Mercator()
ax=pl.axes(projection=m)


ax.set_extent([5,25,-20,-4])

ax.coastlines(color='k')
#ax.add_feature(cartopy.feature.LAND,edgecolor='#697d23',facecolor='#e4f79e')
ax.add_feature(cartopy.feature.BORDERS,edgecolor='k',lw=1)

gl=ax.gridlines(color='#cccccc',draw_labels=1,xlocs=range(5,26,5),ylocs=range(-20,-4,5))
gl.right_labels=False
gl.top_labels=False
gl.bottom_labels=True
gl.left_labels=True

niveis=np.arange(1008,1015.1,0.2)
c=ax.contour(lon,lat,p/100,niveis,transform=ccrs.PlateCarree(),linewidths=0.7,cmap=pl.cm.RdBu)
pl.colorbar(c,shrink=0.7)

# converter coordenadas para o clabel:
pontos_x=np.asarray([10,22,20])
pontos_y=np.asarray([-7,-2,-15])
pp=m.transform_points(ccrs.PlateCarree(),pontos_x,pontos_y)
pontos_x_proj=pp[:,0]
pontos_y_proj=pp[:,1]


pl.clabel(c,manual=zip(pontos_x_proj,pontos_y_proj),colors='k',inline_spacing=0.5,fontsize=8,fmt='%d')
