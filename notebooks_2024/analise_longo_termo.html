def media_anual(t,v):
    anos=[i.year for i in t]
    anos=np.asarray(anos)
    
    v_anual=np.zeros(np.unique(anos).size)
    contador=-1
    for y in np.unique(anos):
        contador+=1
        cond=anos==y
        v_anual[contador]=v[cond].mean()
        
    return np.unique(anos),v_anual

def media_sazonal(t,v):
    anos=[i.year for i in t]
    meses=[i.month for i in tempo]
    
    anos=np.asarray(anos)
    meses=np.asarray(meses)
    
    v_saz=np.zeros(np.unique(anos).size*2)
    t_saz=np.zeros(v_saz.size,dtype=datetime.datetime)
    contador=-1
    for y in np.unique(anos):
        inverno=(anos==y)&((meses==6)|(meses==7)|(meses==8))
        verao=((anos==y)&(meses>8))|((anos==y+1)&(meses<6))

        contador+=1
        v_saz[contador]=v[inverno].mean()
        t_saz[contador]=datetime.datetime(y,7,16)
        
        contador+=1
        v_saz[contador]=v[verao].mean()
        t_saz[contador]=datetime.datetime(y+1,1,16)
        
       
    return t_saz[:-1],v_saz[:-1] # -1 pois ultimo verao fica incompleto!


tsaz,vsaz=media_sazonal(tempo,v)
pl.plot(tsaz,vsaz);

def media_ndias(t,v,nd):
    tn=[]
    vn=[]
    date0=t[0]
    while date0<(t[-1]-datetime.timedelta(days=nd)):
          date1=date0+datetime.timedelta(days=nd)
          cond=(t>=date0)&(t<date1)
          vn+=[v[cond].mean()]
          tn+=[date0+datetime.timedelta(days=nd/2)]
          date0=date1

    tn=np.asarray(tn)
    vn=np.asarray(vn)
        
    return tn,vn

tn,vn=media_ndias(tempo,v,7)
pl.plot(tn,vn);

def clim_diaria(t,v):
    dias=[i.day for i in t]
    meses=[i.month for i in tempo]
    
    dias=np.asarray(dias)
    meses=np.asarray(meses)
    
    # gerar tempo diario para 1 ano:
    td=[datetime.datetime(2001,1,1,12)]
    while td[-1]<datetime.datetime(2001,12,31):
        td+=[td[-1]+datetime.timedelta(days=1)]
    
    td=np.asarray(td)
    vd=np.zeros(td.size)
    
    for i,d in enumerate(td):
        cond=(meses==d.month)&(dias==d.day)
        vd[i]=v[cond].mean()
        
    return td,vd


td,vd=clim_diaria(tempo,v)

fig=pl.figure()

# filtro média movel 30 dias:

ax.plot(td[15:-14],u[15:-14]);
