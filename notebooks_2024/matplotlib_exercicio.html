fig=pl.figure()
fig.set_size_inches([8.5,3])
ax=pl.axes()
ax.plot(a['time'],a['q'],label='rio kwanza')
ax.plot(a['time'],a['q']*0.5,label='rio zaire (m$^3$ s$^{-1}$)')
ax.legend()
ax.set_xlim(date0,date1)

anos=pl.matplotlib.dates.YearLocator()
meses=pl.matplotlib.dates.MonthLocator() 

ax.xaxis.set_major_locator(anos) 
ax.xaxis.set_minor_locator(meses)

yfmt=pl.matplotlib.dates.DateFormatter('%Y-%m')
ax.xaxis.set_major_formatter(yfmt)

adicionar gridlines nos ticks secundário e no eixo x (linhas verticais) a cor verde (green, 'g'), com linha a tracejado e com transparência 0.3 (1=opaco, 0=invisível)¶

fig=pl.figure()
fig.set_size_inches([8.5,3])
ax=pl.axes()
ax.plot(a['time'],a['q'],label='rio kwanza')
ax.plot(a['time'],a['q']*0.5,label='rio zaire (m$^3$ s$^{-1}$)')
ax.legend()
ax.set_xlim(date0,date1)

anos=pl.matplotlib.dates.YearLocator()
meses=pl.matplotlib.dates.MonthLocator() 

ax.xaxis.set_major_locator(anos) 
ax.xaxis.set_minor_locator(meses)

ax.grid(which='major',axis='x',color='r')
ax.grid(which='minor',axis='x',color='g',linestyle='--',alpha=0.3)
