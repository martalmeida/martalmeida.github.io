CARTOPY¶

Cartopy is a Python package designed for geospatial data processing in order to produce maps and other geospatial data analyses

https://scitools.org.uk/cartopy

%matplotlib inline
import pylab as pl

import cartopy.crs as ccrs
import cartopy

m=ccrs.PlateCarree() # proj cilindrica (lon,lat, eixos iguais)

fig=pl.figure()
ax=pl.axes(projection=m)

# linha de costa:
ax.coastlines(color='b',lw=1)

gl=ax.gridlines(color='#cccccc',draw_labels=1,xlocs=range(-180,181,45),ylocs=range(-90,90,45))
gl.right_labels=False
gl.top_labels=False
gl.bottom_labels=True
gl.left_labels=True

fig=pl.figure()
fig.set_size_inches([10,5])
ax0=fig.add_subplot(1,3,1,projection=ccrs.Mollweide())
ax1=fig.add_subplot(1,3,2,projection=ccrs.Orthographic())
ax2=fig.add_subplot(1,3,3,projection=ccrs.Sinusoidal())
for a in [ax0,ax1,ax2]:
    a.add_feature(cartopy.feature.LAND,edgecolor='r',facecolor='#d1ffcd')
    #a.coastlines(color='b',lw=1)
    a.add_feature(cartopy.feature.BORDERS,edgecolor='black',facecolor='none',lw=1)
