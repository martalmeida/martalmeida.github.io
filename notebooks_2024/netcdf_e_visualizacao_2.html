# importar ferramentas do sistema operativo (preciso verificar se existe um ficheiro):
import os

# importar pacote de visualizaao (matplotlib):
import pylab as pl

# importar pacote para fazer e visualizar projecoes:
import cartopy.crs as ccrs 
import cartopy

# import pacote de arrays n-dimensoes:
import numpy as np

# inicializar projecao:
m=ccrs.Robinson()

# matplotlib interactivo (desnecessrio pl.show() depois de cada plot)
pl.ion()

# abrir figura:
fig=pl.figure()

# cria eixos para adicionar elementos grficos:
ax= pl.axes(projection=m)

# adiconar linha de costa:
#ax.coastlines(color='b')

# adicionar linha de costa e/ou regioes de terra:
ax.add_feature(cartopy.feature.LAND,edgecolor='b',facecolor='none')

# adicionar linhas de grelha (gridlines):
gl=ax.gridlines(color='#cccccc',draw_labels=0,xlocs=range(-180,181,30),ylocs=range(-90,91,30))


# carregar dados:
f1='data/tmp/temp_global_era40.npz'

# 1) a partir fo ficheiro f1, se ele existir
if os.path.isfile(f1):
     print('a carregar %s'%f1)
     # abrir ficheiro f1:
     a=np.open(f1)
     # ler as variveis lon, lat e temp que se encontram dentro do ficheiro:
     x=a['lon']
     y=a['lat']
     v=a['temp']
else: 
     # se o ficheiro f1 nao existir, ler directamente a partir do servidor de dados
     # (servidor thredds, que permite usar variveis como se estivessem num ficheiro local netcdf):
     f='http://apdrc.soest.hawaii.edu:80/dods/public_data/ERA-40/monthly-pressure'
     import netCDF4
     # abrir ficheiro netcdf:
     nc=netCDF4.Dataset(f)
     print(nc)
     print(list(nc.variables))
     print(nc.variables['air'])
        
     # ler varivel air, de 4 dimensoes (tempo, nivel vertical, lat e lon) no tempo indice 0 e nvel
     # vertical indice 0:
     v=nc.variables['air'][0,0,:,:]
    
     # ler coordenadas longitude e latitude:
     x=nc.variables['lon'][:]
     y=nc.variables['lat'][:]
     #print(nc.variables['lon'])

     # gravar variveis no formato do numpy (npz) no ficheiro f1
     print('a gravar %s'%f1)
     np.savez(f1,lon=x,lat=y,temp=v)


# visualizar temperatura na projecao definida antes. A opao transform=ccrs.PlateCarree()
# indica que as coordenadas x,y dados estao em graus (ou seja, sao dados nao projectados):
ax.pcolormesh(x,y,v,transform=ccrs.PlateCarree())
