def ler_boia(file):
    '''Ler dados (tempos e temperaturas)'''
    # ler tudo.
    linhas=open(file).readlines()
    n=len(linhas) # numero tital de linhas
    n_header=13 # numero linhas do cabeçalho
    
    # array (vector) para guardar a variável de interesse (temperatura)
    # este array vai conter dados numéricos
    v=np.zeros(n-n_header)
    
    # array para guardar os tempos
    # este array Nao vai ter dados numéricos mas sim data, ie, objectos do tipo dat
    # daí necessário formecer o dtype=datetime.datetime
    time=np.zeros(n-n_header,dtype=datetime.datetime)
    
    # para todas as linha depois do cabeçalho
    for i in range(n_header,n):
        # dividir cada linha pelo caracter ","
        # antes de dividir, fazer o strip(), ie, eliminar estaços (no inicio e fim)
        # e caracteres de fim de linha (\n e \t \r ?!)
        tmp=linhas[i].strip().split(',')
        
        # ano será o primeiro elemento convertido de string para inteiro
        ano=int(tmp[0])
        # mês será o segundo
        mes=int(tmp[1])
        # etc
        dia=int(tmp[2])
        hora=int(tmp[3])
        mnt=int(tmp[4])
        # criar objecto datetime.datetime
        t=datetime.datetime(ano,mes,dia,hora,mnt)
        # guardar no array criado acima (time)
        time[i-n_header]=t
        # ler variável (ultimo elemento do split), ie, converter para float (numero
        # do último tmp. Mas este pode ter o valor "NULL". Se assim for, guardar co
        if tmp[-1]=='NULL':
            v[i-n_header]=99
        else:
            v[i-n_header]=float(tmp[-1])
            
    # caso haja valores 99 no array v, vou mascará-los de forma a que nao apareçam
    # nem entrem em possíveis cálculos
    v=np.ma.masked_where(v==99,v)
    # retornar a temperatura e o tempo
    return v,time

def ler_info(file):
    '''Ler informaçoes do cabeçalho'''
    # lê todas as linhas numa lista
    linhas=open(file).readlines()
    n=len(linhas) # numero de linhas (tamanha da lista linhas)
    n_header=13   # numero linhas cabeçalho
    
    # ler primeira linha, com nome da estaçao (comçear na coluna 2)
    nome=linhas[0].strip()[2:]
    
    # para todas as linha do cabeçalho, verificar se a linha começa por /longitude
    for i in range(n_header):
        if linhas[i].startswith('/longitude'):
            # se começa por /longitude, dividir a linha (split) pelo caracter "="
            # e o valor está no indice 1 desta divisao; está como string, preciso
            # converter para float (numero real)
            tmp=linhas[i].split('=')
            lon=float(tmp[1])
            
        elif linhas[i].startswith('/latitude'):
            # se começa por /latitude, dividir a linha (split) pelo caracter "="
            # e o valor está no indice 1 desta divisao; está como string, preciso
            # converter para float (numero real)
            tmp=linhas[i].split('=')
            lat=float(tmp[1])
            
    # devolver os três dados que procuravamos
    return nome,lon,lat

    n=len(linhas)
    n_header=1
    
    v=np.zeros(n-n_header)
    time=np.zeros(n-n_header,dtype=datetime.datetime)

        tmp0=tmp[0].split('-')

        mes=int(tmp0[1])
        dia=int(tmp0[2])

        # criar objecto datetime.datetime
        t=datetime.datetime(ano,mes,dia)
        time[i-n_header]=t
        
        if tmp[-1]=='NULL':
            v[i-n_header]=99
        else:
            v[i-n_header]=float(tmp[-1])
            
    v=np.ma.masked_where(v==99,v)
    return v,time
