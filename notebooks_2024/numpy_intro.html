Numpy¶

Numpy adds n-d arrays to Python. It allows greater flexibility and performance that the native iterables (like lists and tuples).

Almost all the Python scientific tools are based on numpy.

def rot2d(x,y,ang):
    ang=ang*np.pi/180
    
    V=[x,y]
    M=[[np.cos(ang),-np.sin(ang)],
       [np.sin(ang),np.cos(ang)]]
    
    V=np.asarray(V)
    M=np.asarray(M)

def local_max_min(y):
    n=y.size
    iMax=[]
    iMin=[]
    iConst=[]

    # start:
    if y[0]>=y[1]: iMax+=[0]
    if y[0]<=y[1]: iMin+=[0]
    if y[0]==y[1]: iConst+=[0]

    # internal:
    for i in range(1,n-1):
        if y[i-1] <= y[i] and y[i+1] <= y[i]: iMax+=[i]
        if y[i-1] >= y[i] and y[i+1] >= y[i]: iMin+=[i]
        if y[i-1] == y[i] == y[i+1]: iConst+=[i]

    # end:
    if y[-1]>=y[-2]: iMax+=[n-1]
    if y[-1]<=y[-2]: iMin+=[n-1]
    if y[-1]==y[-2]: iConst+=[n-1]

    return iMin,iMax,iConst
