time series¶

Matplotlib can deal with dates as arrays of datetime objects, e.g.:

let's make this look better...

In [12]:
import datetime
date0=datetime.datetime(2013,1,1)
date1=datetime.datetime(2016,1,1)

fig,ax=pl.subplots(1)
fig.set_size_inches((9,3))
ax.plot(t,q)

ax.set_xlim(date0,date1)
ax.set_ylim(0,5000)
ax.xaxis.grid(1)
ax.yaxis.grid(1)

# x ticks and labels:
years=pl.matplotlib.dates.YearLocator()
months=pl.matplotlib.dates.MonthLocator()
yfmt=pl.matplotlib.dates.DateFormatter('%Y')

ax.xaxis.set_major_locator(years)
ax.xaxis.set_tick_params(pad=10)
ax.xaxis.set_major_formatter(yfmt)
ax.xaxis.set_minor_locator(months)

d0=datetime.datetime(2014,1,1)
d1=datetime.datetime(2015,1,1)
yl=ax.get_ylim()
ax.fill([d0,d1,d1,d0],[yl[0],yl[0],yl[1],yl[1]],'#fff5d1',zorder=0.9)

import matplotlib.transforms as transforms
# the x coords of this transformation are data, and the
# y coord are axes (0 to 1)
trans = transforms.blended_transform_factory(
    ax.transData, ax.transAxes)

for y in range(2013,2016):
    for m in [3,6,9,12]:
        d=datetime.datetime(y,m-2,16)
        ax.text(d,-0.03,d.strftime('%b').upper(),transform=trans,
                fontsize=7,va='top',ha='center',color='g')
        d=datetime.datetime(y,m,1)
        ax.plot([d,d],ax.get_ylim(),'g--',lw=0.5,alpha=0.7)

transforms¶

In the previous time series example we used a transform to place text using data at x and axes coords at y. Let's play a little with this:

In [13]:
ax=pl.axes()
pl.plot(q[:30])

# put some label at the center of the axes:
pl.text(.5,.5,'CENTRE',ha='center',va='center',transform=ax.transAxes,fontsize=12)
# or at upper left corner:
pl.text(-0.1,1,'a) DISCHARGE (m$^3$ s$^{-1}$)',ha='left',va='bottom',
        transform=ax.transAxes,fontsize=12)

# label at data x,y:
xo=10
yo=300
pl.plot(xo,yo,'r+')
pl.text(xo,yo,'(%d,%d)'%(xo,yo))

# label ax mixed coordinates (end of x axis, y=500):
trans = transforms.blended_transform_factory(
    ax.transAxes, ax.transData)
pl.text(1,500,'LABEL',ha='right',va='center',transform=trans,fontsize=12);
