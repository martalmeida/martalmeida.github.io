Basemap¶

Library for plotting 2D data on maps in Python

In [1]:
import numpy as np
%matplotlib inline
import pylab as pl
from mpl_toolkits.basemap import Basemap

xlim=-20,20
ylim=35,55.0001 # 0001 is to show the parallel 55N

m = Basemap(projection='merc',llcrnrlat=ylim[0],urcrnrlat=ylim[1],
            llcrnrlon=xlim[0],urcrnrlon=xlim[1],resolution='i')

def draw():
    m.drawcoastlines(color='#923192')
    #m.drawrivers(color='r')
    m.drawcountries(color='#923192')
    m.fillcontinents(color='#d4d4d4')
    #m.drawmapboundary(fill_color='#f8f8f8')
    m.drawmeridians(range(xlim[0],xlim[1]+1,5),labels=[0,0,0,1],linewidth=0.5)
    m.drawparallels(np.arange(ylim[0],ylim[1]+1,5),labels=[1,0,0,0],linewidth=0.5)
    
draw()
