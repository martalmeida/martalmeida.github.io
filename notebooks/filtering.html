def calc_lowpass(v,dt_hours=1,ndays=90,padtype='odd'):
    dt=dt_hours
    from scipy import signal
    sampl       = 1./dt
    nyq         = sampl/2.
    cutoff      = 1./(ndays*24)
    filterorder = 5
    filterorder = 4

    b,a = signal.filter_design.butter(filterorder,cutoff/nyq,btype='low')
    y = signal.filtfilt(b,a,v,padtype=padtype)
    return y

def plt(ax,t,u,v,ndays,qkey=False):
    uf=calc_lowpass(u,24,ndays)
    vf=calc_lowpass(v,24,ndays)
    s=5
    n=3
    w=0.001
    cond=vf[::n]>0
    hp=ax.quiver(t[::n][cond],np.zeros(t.size)[::n][cond],uf[::n][cond],vf[::n][cond],
                scale=s,width=w,color='r')
    hn=ax.quiver(t[::n][~cond],np.zeros(t.size)[::n][~cond],uf[::n][~cond],vf[::n][~cond],
                scale=s,width=w)
    
    if qkey:
        pl.quiverkey(hn,0.05,0.7,0.2,'20 cm s$^{-1}$')

    ax.set_yticks(())
    for k in ax.spines:##('top','left','right'):
        ax.spines[k].set_visible(0)
