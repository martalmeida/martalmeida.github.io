import netCDF4
%matplotlib inline
import pylab as pl
import numpy as np

nc=netCDF4.Dataset('data/vict/grd_vict.nc4')
x=nc.variables['lon_rho']
y=nc.variables['lat_rho']

def fronteira(v):
    W=v[:,0]
    E=v[:,-1][::-1]
    S=v[0][::-1]
    N=v[-1]
    return np.concatenate((W,N,E,S))

xb=fronteira(x)
yb=fronteira(y)
pl.fill(xb,yb,'g',alpha=.2)
n=15
pl.plot(x[:,::n],y[:,::n],'b',alpha=.5);
pl.plot(x[10::n,:].T,y[10::n,:].T,'b',alpha=.5);

from mpl_toolkits.basemap import Basemap

# projecção:
if nc.MAP_PROJ == 1:
    proj = 'lcc'
elif nc.MAP_PROJ == 3:
    proj = 'merc'

# resolução (coasts, rios, fronteiras, ...)
res='l'

lat_1 = nc.TRUELAT1
lat_2 = nc.TRUELAT2
lon_0 = nc.CEN_LON
lat_0 = nc.CEN_LAT
    
m = Basemap(projection=proj, resolution=res, width=7e6,height=6e6,
            lat_1=lat_1, lat_2=lat_2, lat_0=lat_0, lon_0=lon_0)
    

mer=np.arange(-100,100,10)
par=np.arange(-80,90,10)
m.drawparallels(par,  labels=[1,0,0,0],linewidth=.5,
                   color='#4d4d4d',dashes=[1,4])
m.drawmeridians(mer, labels=[0,0,0,1],linewidth=.5,
                   color='#4d4d4d',dashes=[1,4])

m.fillcontinents(color='#cccccc')
m.drawcountries(color='w')

Xb1,Yb1=m(xb1,yb1)
pl.plot(Xb1,Yb1)
Xb,Yb=m(xb,yb)
pl.plot(Xb,Yb);
